                <h2>Import Exam Programs (CSV)</h2>

                    <p>Upload a CSV file with the required columns. The import will run in batches to prevent server timeouts.</p>
                    <p><strong>Instructions:</strong></p>
                    <ol style="list-style:decimal; padding-left:20px;">
                        <li>Download the template to ensure your file has the correct structure.</li>

                        </p>

                    </form>
                </div>

                    <p>Importing... Please do not close this window.</p>
                    <div style="background:#eee; border-radius:4px; padding:2px; margin-bottom:1rem;">
                        <div id="mco-progress-bar" style="background:#0073aa; height:20px; width:0%; border-radius:2px; transition: width 0.3s ease;"></div>
                    </div>
                    <div id="mco-import-feedback" style="max-height: 200px; overflow-y: auto; background: #f9f9f9; border: 1px solid #ccc; padding: 10px; border-radius: 4px; margin-bottom: 1rem;"></div>
                    <button id="mco-cancel-import" class="button">Cancel Import</button>
                </div>
            </div>
        </div>
        <script>
        jQuery(document).ready(function($){
            let totalRows = 0;
            let processedRows = 0;
            let importTransientKey = '';
            let isCancelled = false;

            $('#mco-import-form').on('submit', function(e){
                e.preventDefault();
                $('#mco-import-initial-view').hide();
                $('#mco-import-progress-view').show();
                $('#mco-import-feedback').html('');
                isCancelled = false;
                
                var formData = new FormData(this);
                formData.append('action', 'mco_import_csv_start');

                $.ajax({
                    url: ajaxurl, type: 'POST', data: formData, processData: false, contentType: false,
                    success: function(response) {
                        if (response.success) {
                            totalRows = response.data.total_rows;
                            importTransientKey = response.data.transient_key;
                            processedRows = 0;
                            $('#mco-import-feedback').text('Found ' + totalRows + ' rows to import. Starting process...');
                            processBatch(1);
                        } else {
                            $('#mco-import-feedback').html('<p style="color:red;"><strong>Error:</strong> ' + response.data + '</p>');

                    error: function() {
                        $('#mco-import-feedback').html('<p style="color:red;"><strong>Error:</strong> An unexpected server error occurred during setup.</p>');
                    }
                });
            });

            function processBatch(batchNumber) {
                if (isCancelled) return;

                $.ajax({
                    url: ajaxurl, type: 'POST',
                    data: {
                        action: 'mco_import_csv_batch',
                        nonce: '<?php echo wp_create_nonce("mco_import_batch_nonce"); ?>',
                        transient_key: importTransientKey,
                        batch: batchNumber
                    },
                    success: function(response) {
                        if (isCancelled) return;
                        if (response.success) {
                            processedRows += response.data.imported;
                            let percent = totalRows > 0 ? Math.min(100, Math.round((processedRows / totalRows) * 100)) : 0;
                            $('#mco-progress-bar').css('width', percent + '%');
                            $('#mco-import-feedback').text('Processed ' + processedRows + ' of ' + totalRows + ' rows...');

                            if (response.data.done) {
                                $('#mco-import-feedback').html('<p style="color:green;"><strong>Import complete!</strong> Processed ' + processedRows + ' rows. The page will now reload.</p>');
                                setTimeout(function(){ location.reload(); }, 2000);
                            } else {
                                processBatch(batchNumber + 1);
                            }
                        } else {
                             $('#mco-import-feedback').append('<p style="color:red;"><strong>Batch Error:</strong> ' + response.data + '</p>');
                        }
                    },
                    error: function() {
                         if (isCancelled) return;
                        $('#mco-import-feedback').append('<p style="color:red;"><strong>Error:</strong> A server error occurred while processing a batch.</p>');
                    }
                });
            }

            $('#mco-cancel-import').on('click', function(){
                if (!isCancelled) {
                    isCancelled = true;
                    $(this).prop('disabled', true).text('Cancelling...');
                    $.ajax({
                        url: ajaxurl, type: 'POST',
                        data: {
                            action: 'mco_import_csv_cancel',
                            nonce: '<?php echo wp_create_nonce("mco_import_cancel_nonce"); ?>',
                            transient_key: importTransientKey,
                        },
                        success: function(response) {
                             $('#mco-import-feedback').html('<p style="color:orange;"><strong>Import cancelled.</strong> The page will reload.</p>');
                             setTimeout(function(){ location.reload(); }, 2000);
                        }
                    });
                }
            });
        });
        </script>
        <?php
    }
}

if (!function_exists('mco_handle_csv_template_download')) {
    function mco_handle_csv_template_download() {
        if (!current_user_can('manage_options')) {
            wp_die('Permission denied.');
        }
        header('Content-Type: text/csv');
        header('Content-Disposition: attachment; filename="exam_program_template.csv"');
        $header = ['post_title', 'post_content', 'post_name', '_mco_question_source_url', 'exam_practice_questions', 'exam_practice_duration', 'exam_cert_questions', 'exam_cert_duration', 'exam_pass_score', '_mco_certification_exam_sku', '_mco_bundle_sku'];

    }
}

if (!function_exists('mco_handle_ajax_import_start')) {
    function mco_handle_ajax_import_start() {
        if (!check_admin_referer('mco_import_csv_nonce', 'mco_import_csv_nonce') || !current_user_can('manage_options') || !isset($_FILES['mco_csv_file'])) {
            wp_send_json_error('Security check failed or file not found.');
        }

        $file = $_FILES['mco_csv_file'];
        if ($file['error'] !== UPLOAD_ERR_OK) { wp_send_json_error('File upload error.'); }

        $handle = fopen($file['tmp_name'], 'r');
        $header = fgetcsv($handle);
        $expected_header = ['post_title', 'post_content', 'post_name', '_mco_question_source_url', 'exam_practice_questions', 'exam_practice_duration', 'exam_cert_questions', 'exam_cert_duration', 'exam_pass_score', '_mco_certification_exam_sku', '_mco_bundle_sku'];
        
        if (array_map('trim', $header) !== $expected_header) {
            fclose($handle);
            wp_send_json_error('Invalid CSV header. Please use the provided template. Expected: ' . implode(', ', $expected_header));
        }

        $total_rows = 0;
        while (fgetcsv($handle) !== false) { $total_rows++; }
        fclose($handle);

        $upload_dir = wp_upload_dir();
        $temp_filename = wp_unique_filename($upload_dir['path'], $file['name']);
        $temp_filepath = $upload_dir['path'] . '/' . $temp_filename;

        if (!move_uploaded_file($file['tmp_name'], $temp_filepath)) {
            wp_send_json_error('Could not move uploaded file.');
        }

        $transient_key = 'mco_import_' . md5($temp_filepath);
        set_transient($transient_key, ['filepath' => $temp_filepath, 'total_rows' => $total_rows], HOUR_IN_SECONDS);

        wp_send_json_success(['total_rows' => $total_rows, 'transient_key' => $transient_key]);
    }
}

if (!function_exists('mco_handle_ajax_import_batch')) {
    function mco_handle_ajax_import_batch() {
        check_ajax_referer('mco_import_batch_nonce', 'nonce');
        if (!current_user_can('manage_options')) { wp_send_json_error('Permission denied.'); }

        $transient_key = sanitize_key($_POST['transient_key']);
        $import_data = get_transient($transient_key);
        if (false === $import_data) { wp_send_json_error('Import session expired or invalid.'); }

        $batch_size = 20;
        $batch_number = intval($_POST['batch']);
        $offset = ($batch_number - 1) * $batch_size;
        
        $result = mco_import_exam_programs_batch($import_data['filepath'], $offset, $batch_size);

        $done = ($offset + $result['imported']) >= $import_data['total_rows'] || $result['imported'] < $batch_size;
        if ($done) {
            unlink($import_data['filepath']);
            delete_transient($transient_key);
        }

        wp_send_json_success(['imported' => $result['imported'], 'done' => $done]);
    }
}

if (!function_exists('mco_import_exam_programs_batch')) {
    function mco_import_exam_programs_batch($csv_file_path, $offset, $limit) {
        $imported = 0;
        $header = ['post_title', 'post_content', 'post_name', '_mco_question_source_url', 'exam_practice_questions', 'exam_practice_duration', 'exam_cert_questions', 'exam_cert_duration', 'exam_pass_score', '_mco_certification_exam_sku', '_mco_bundle_sku'];
        
        $handle = fopen($csv_file_path, 'r');
        fgetcsv($handle); // Skip header

        for ($i = 0; $i < $offset; $i++) {
            if (fgetcsv($handle) === false) break;
        }

        while ($imported < $limit && ($data = fgetcsv($handle)) !== FALSE) {
            $row = array_combine($header, array_pad($data, count($header), null));
            $post_id = wp_insert_post([
                'post_title'   => sanitize_text_field($row['post_title']),
                'post_content' => wp_kses_post($row['post_content']),
                'post_name'    => sanitize_title($row['post_name']),
                'post_type'    => 'mco_exam_program',
                'post_status'  => 'publish'
            ]);
            if (!is_wp_error($post_id)) {
                update_post_meta($post_id, '_mco_question_source_url', esc_url_raw($row['_mco_question_source_url']));
                update_post_meta($post_id, '_mco_certification_exam_sku', sanitize_text_field($row['_mco_certification_exam_sku']));
                update_post_meta($post_id, '_mco_bundle_sku', sanitize_text_field($row['_mco_bundle_sku']));
                $taxonomies = ['exam_practice_questions', 'exam_practice_duration', 'exam_cert_questions', 'exam_cert_duration', 'exam_pass_score'];
                foreach ($taxonomies as $tax) {
                    if (!empty($row[$tax])) wp_set_object_terms($post_id, intval($row[$tax]), $tax);
                }
                $imported++;
            }
        }
        fclose($handle);
        return ['imported' => $imported];
    }
}

if (!function_exists('mco_handle_ajax_import_cancel')) {
    function mco_handle_ajax_import_cancel() {
        check_ajax_referer('mco_import_cancel_nonce', 'nonce');
        if (!current_user_can('manage_options')) { wp_send_json_error('Permission denied.'); }

        $transient_key = sanitize_key($_POST['transient_key']);
        $import_data = get_transient($transient_key);
        if ($import_data && isset($import_data['filepath'])) {
            unlink($import_data['filepath']);
            delete_transient($transient_key);
        }
        wp_send_json_success('Import cancelled and cleaned up.');
    }
}
